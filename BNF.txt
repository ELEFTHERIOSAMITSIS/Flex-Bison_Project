<dividend> ::= <FLOAT> 
<winners> ::= <INTEGER>
<distributed> ::= <FLOAT>
<jackpot> ::= <FLOAT>
<fixed> ::= <FLOAT>
<categoryType> ::= 0|1
<gameType> ::= <STRING>
< minimumDistributed > ::= <FLOAT>
<columns> ::= <INTEGER>
<wagers> ::= <INTEGER>
<addOn> ::= []
<content> ::= <gameId> <drawId> <drawTime> <status> <drawBreak> <visualDraw> <pricePoints>  <prizeCategories> <wagerStatistics>|
